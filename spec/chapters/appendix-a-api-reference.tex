\begin{longtable}{|l|l|l|p{6cm}|}

\hline

POST & /api/v1/auth/oauth/:provider & No & OAuth login \\
POST & /api/v1/auth/refresh & No & Refresh access token \\
POST & /api/v1/auth/logout & Yes & Logout user \\
POST & /api/v1/auth/verify-email & No & Verify email address \\
POST & /api/v1/auth/reset-password & No & Request password reset \\
POST & /api/v1/auth/reset-password/confirm & No & Confirm password reset \\
\hline
\end{longtable}
