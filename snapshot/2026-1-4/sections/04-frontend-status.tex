\section{Frontend Status}

\subsection{Frontend Framework}
\begin{itemize}
    \item \textbf{Framework}: Next.js 14.0.4 (App Router)
    \item \textbf{Language}: TypeScript
    \item \textbf{UI Library}: React 18.2.0
    \item \textbf{Styling}: Tailwind CSS 3.4.0
    \item \textbf{Authentication}: NextAuth.js 4.24.5
    \item \textbf{Real-time}: Socket.io Client 4.6.1
    \item \textbf{Port}: 3000 (development)
\end{itemize}

\subsection{WebSocket Hooks}

\subsubsection{useGameWebSocket}
\begin{itemize}
    \item Connects to \texttt{/games} namespace
    \item Handles: move-made, game-update, game-finished
    \item Manages connection state
    \item Auto-reconnection
\end{itemize}

\subsubsection{useChatWebSocket}
\begin{itemize}
    \item Connects to \texttt{/chat} namespace
    \item Handles: new-message
    \item Provides sendMessage function
    \item Manages connection state
\end{itemize}

\subsubsection{useVotingWebSocket}
\begin{itemize}
    \item Connects to \texttt{/voting} namespace
    \item Handles: voting-started, vote-update, voting-ended
    \item Provides submitVote function
    \item Manages connection state
\end{itemize}

\subsection{API Client}
\begin{itemize}
    \item Centralized fetch wrapper (\texttt{lib/api.ts})
    \item Type-safe API calls
    \item Automatic token management
    \item Error handling
    \item API functions for:
    \begin{itemize}
        \item Auth (login, register, getMe)
        \item Games (list, get, create, move, resign, draw)
        \item Ratings (getMyRatings, leaderboard, history)
        \item Voting (start, vote, status, end)
        \item Chat (getMessages, sendMessage)
    \end{itemize}
\end{itemize}
