\subsection{Technology Stack}
\begin{itemize}
    \item \textbf{Frontend}: Next.js 14, React 18, TypeScript, Tailwind CSS, NextAuth.js
    \item \textbf{Backend}: Nest.js 10, TypeScript, Prisma ORM, Socket.io
    \item \textbf{Database}: Supabase (PostgreSQL)
    \item \textbf{Authentication}: JWT, NextAuth.js
    \item \textbf{Real-time}: Socket.io (WebSocket)
    \item \textbf{Chess Engine}: chess.js
    \item \textbf{Monorepo}: Turborepo, pnpm workspaces
\end{itemize}
